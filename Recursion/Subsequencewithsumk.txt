public class SubsequenceSumK {

    static void printSubsequence(int index, int[] arr, List<Integer> curr, int sum, int k) {

        // Base case
        if (index == arr.length) {
            if (sum == k) {
                System.out.println(curr);
            }
            return;
        }

        // Include current element
        curr.add(arr[index]);
        printSubsequence(index + 1, arr, curr, sum + arr[index], k);

        // Backtrack
        curr.remove(curr.size() - 1);

        // Exclude current element
        printSubsequence(index + 1, arr, curr, sum, k);
    }
}
