 Ab ye problem ka sol jb maine likha toh ye tha 

    public int lengthOfLongestSubstring(String s) {
        Set<Character> set = new HashSet<>();

        int l = 0;
        int r = 0;
        int maxlen = 0;
        while(r<s.length()){
            char c = s.charAt(r);

            if(set.contains(c)){
                set.remove(s.charAt(l));
                l++;
            }
            set.add(c);
            maxlen = Math.max(maxlen , r - l + 1);
            r++;
        }
        return maxlen;
    }
